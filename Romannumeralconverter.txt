var a=[1000,900,500,400,100,90,50,40,10,9,5,4,1];
var b=["M","CM","D","CD","C","XC","L","XL","X","IX","V","IV","I"];
function convertToRoman(num) {
  var i=0;
  var ans="";
  while(num!=0)
  {
    while(num>=a[i])
    {
      ans+=b[i];
      num-=a[i];
    }
    i++;
  }
 return ans;
}

convertToRoman(36);
